-- WIN POWERSHELL --

mkdir build
cd build
cmake .. -DCMAKE_BUILD_TYPE=Release `
    -DCMAKE_CXX_FLAGS="-static-libgcc -static-libstdc++" `
    -DCMAKE_EXE_LINKER_FLAGS="-static"
cmake --build . --config Release
cmake --install . --prefix ../dist --config Release
